CREATE TABLE Contractor (
ContractorID INTEGER NOT NULL,
FirstName VARCHAR(20) NOT NULL,
LastName VARCHAR(20) NOT NULL,
MiddleInitial CHAR(1) NULL,
HouseNumber INTEGER NULL,
Street VARCHAR(50) NULL,
CityCounty VARCHAR(40) NULL,
StateAbb CHAR(2) NULL,
CountryAbb CHAR(2) NULL,
ZipCode CHAR(5) NULL,
Fee DECIMAL(8,2) NULL,
LastUpdatedBy VARCHAR(20) NULL,
LastUpdated DATE NULL,
CONSTRAINT PK_ContractorID PRIMARY KEY (ContractorID),
CONSTRAINT FK_ContractorState FOREIGN KEY (StateAbb) references HomeState,
CONSTRAINT FK_ContractorCountry FOREIGN KEY (CountryAbb) references Country);

CREATE TABLE Driver (
DriverID INTEGER NOT NULL,
FirstName VARCHAR(20) NOT NULL,
LastName VARCHAR(20) NOT NULL,
MiddleInitial CHAR(1) NULL,
HouseNumber INTEGER NULL,
Street VARCHAR(50) NULL,
CityCounty VARCHAR(40) NULL,
StateAbb CHAR(2) NULL,
CountryAbb CHAR(2) NULL,
ZipCode CHAR(5) NULL,
DateOfBirth DATE NULL,
CDL VARCHAR(40) NULL,
CDLDate DATE NULL,
LastUpdatedBy VARCHAR(20) NULL,
LastUpdated DATE NULL,
CONSTRAINT PK_DriverID PRIMARY KEY (DriverID),
CONSTRAINT FK_DriverState FOREIGN KEY (StateAbb) references HomeState,
CONSTRAINT FK_DriverCountry FOREIGN KEY (CountryAbb) references Country);

CREATE TABLE Equipment (
ID INTEGER NOT NULL,
VinNumber VARCHAR(40) NULL,
Make VARCHAR(35) NULL,
Model VARCHAR(30) NULL,
EquipmentYear INTEGER NULL,
PriceAcquired DECIMAL(8,2) NULL,
LicenseNumber VARCHAR(10) NULL,
DriverID INTEGER NULL,
LastUpdatedBy VARCHAR(20) NULL,
LastUpdated DATE NOT NULL,
CONSTRAINT PK_EquipmentID PRIMARY KEY (ID),
CONSTRAINT FK_EquipmentDriverID FOREIGN KEY (DriverID) references Driver);

CREATE TABLE DriverContractor (
ContractorID INTEGER NOT NULL,
DriverID INTEGER NOT NULL,
HireDate DATE NULL,
TerminationDate DATE NULL,
Salary DECIMAL(10,2) NULL,
LastUpdatedBy VARCHAR(20) NULL,
LastUpdated DATE NULL,
CONSTRAINT PK_DCID PRIMARY KEY (ContractorID, DriverID),
CONSTRAINT FK_DCContractorID FOREIGN KEY (ContractorID) references Contractor,
CONSTRAINT FK_DCDriverID FOREIGN KEY (DriverID) references Driver);
